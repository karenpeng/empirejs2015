  function TaskManager() {
    this.tasks = [];
  }

  TaskManager.prototype.executeTasks = function (tasks) {
    this.tasks = tasks;
    this._execute();
  };

  TaskManager.prototype._execute = function () {
    if (!this.tasks.length) {
      return;
    } else {
      var ta = this.tasks.shift();
      var direction = ta[0];
      this.move(direction);
    }
  };
